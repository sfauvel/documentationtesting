= Source code using a minimal builder


[red]##_Usage_##
[source,java,indent=0]
----
        output = formatter.sourceCodeBuilder()
                .source(
                        "public int add(int a, int b) {\n" +
                                "   int result = a + b;\n" +
                                "   return result;\n" +
                                "}")
                .build()
        ;
----

   int result = a + b;
   return result;
}
