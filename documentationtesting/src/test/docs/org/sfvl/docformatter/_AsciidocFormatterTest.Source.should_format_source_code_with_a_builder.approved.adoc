= Source code using a builder


[red]##_Usage_##
[source,java,indent=0]
----
            output = formatter.sourceCodeBuilder("groovy")
                    .title("Source code")
                    .indent(4)
                    .source(
                            "public int add(int a, int b) {\n" +
                                    "   int result = a + b;\n" +
                                    "   return result;\n" +
                                    "}")
                    .build()
            ;
----

[red]##_Render_##

.Source code
[source,groovy,indent=4]
----
public int add(int a, int b) {
   int result = a + b;
   return result;
}
----
