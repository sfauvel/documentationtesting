.Source code to extract

 * Comment of the class.

public class EnumWithCommentToExtract {

    /**
     * Comment of the enum.

         * First enum with comment.
         */
        FirstEnum,
        /** Second enum with comment. */
        SecondEnum,
        ThirdEnum;

        /**
         * A method in enum.

        }
    }

}
